For loop
num = int(input("enter a value : " ))
value = int(input("enter a value : " ))

for i in range(1,num+1):
    print(f"{value}*{i} = {i*value}")

Loop Break
num = int(input("enter a range : " ))
value = int(input("enter a value : " ))
stop = int(input("enter a number : " ))

for i in range(1,num+1):
    print(f"{value}*{i} = {i*value}")
    
    if i == stop:
        break
-----------------------------------------------------------------------------------------------------
loops COntinue
num = int(input("enter a range : " ))
value = int(input("enter a value : " ))
skip = int(input("enter a number : " ))

for i in range(1,num+1):
    if i == skip:
        continue
        
    print(f"{value}*{i} = {i*value}")
